Apps Script — Drive folder → ingest AWB batches 
// Time-trigger this (e.g. every 5 min). AWB files in the folder must be Google
// Sheets (xlsx uploads: turn on "convert uploads" or Drive-convert them first).
const INGEST_URL = '{{ url_for('api_awb_ingest', _external=True) }}';
const TOKEN      = '{{ cfg.AWB_API_TOKEN or 'PASTE_TOKEN_HERE' }}';
const FOLDER_ID  = 'YOUR_DRIVE_FOLDER_ID';

function pushNewAwbFiles() {
  const props = PropertiesService.getScriptProperties();
  const since = Number(props.getProperty('lastRun') || 0);
  const files = DriveApp.getFolderById(FOLDER_ID).getFiles();
  while (files.hasNext()) {
    const f = files.next();
    if (f.getLastUpdated().getTime() <= since) continue;
    const data = SpreadsheetApp.open(f).getSheets()[0].getDataRange().getValues();
    const headers = data.shift().map(h => String(h).trim());
    const rows = data.filter(r => r.join('')).map(r => {
      const o = {}; headers.forEach((h, i) => o[h] = r[i]); return o;
    });
    UrlFetchApp.fetch(INGEST_URL, {
      method: 'post', contentType: 'application/json',
      headers: { 'X-API-Key': TOKEN }, muteHttpExceptions: true,
      payload: JSON.stringify({ rows: rows })
    });
  }
  props.setProperty('lastRun', String(Date.now()));
}
Apps Script — status sheet → mark delivered 
// Add as an installable "on edit" trigger on your AWB status sheet.
// Set AWB_COL / STATUS_COL to your columns (1 = A, 2 = B, ...).
const STATUS_URL = '{{ url_for('api_awb_status', _external=True) }}';
const TOKEN      = '{{ cfg.AWB_API_TOKEN or 'PASTE_TOKEN_HERE' }}';
const AWB_COL    = 1;
const STATUS_COL = 2;

function onEditAwbStatus(e) {
  const sh = e.range.getSheet(), row = e.range.getRow();
  if (row === 1) return;
  const status = String(sh.getRange(row, STATUS_COL).getValue());
  if (!/deliver/i.test(status)) return;
  const awb = sh.getRange(row, AWB_COL).getValue();
  UrlFetchApp.fetch(STATUS_URL, {
    method: 'post', contentType: 'application/json',
    headers: { 'X-API-Key': TOKEN }, muteHttpExceptions: true,
    payload: JSON.stringify({ awb: awb, status: 'delivered' })
  });
}
